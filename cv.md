function duplicateCount(text){
  const _ = require("lodash");
  let str = text.toLowerCase();
  let col = str.length;
  let morin = [];
  let iter;
  for(let i=0;i<col;i++){
    iter = 0;
    for(let j=0;j<col;j++){
    if(str[i]===str[j]) {iter++; }
    if(iter===2){
      morin.push(str[i]); iter++;
    }
      }
  }
  morin.sort();
  let sum = _.sortedUniq(morin);
  return sum.length;
}
